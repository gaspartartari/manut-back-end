package com.manut.api;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.util.List;
import java.util.Set;
import java.time.LocalDateTime;

public class TimeZoneTest {
    public static void main(String[] args) {
        ZoneId zoneIdSP = ZoneId.of("America/Sao_Paulo");
        LocalDateTime ldt = LocalDateTime.now();
        ZonedDateTime zdt = ZonedDateTime.of(ldt, zoneIdSP);
        ZonedDateTime utc = zdt.withZoneSameInstant(ZoneId.of("UTC"));

        System.out.println("Local Date-Time: " + ldt);
        System.out.println("Zoned Date-Time in Sao Paulo: " + zdt);
        System.out.println("Current Offset in Sao Paulo: " + zdt.getOffset());
        System.out.println("Current Offset in UTC: " + utc);

        Set<String> zones = ZoneId.getAvailableZoneIds();
        for (String z : zones){
            System.out.println(z);
        }
    }

   
}
